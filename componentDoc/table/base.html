# 参数说明

//参数说明
tableColumns:[
    {
      label:'姓名',//列名
      prop:'prop',//列属性
      width:'120',//列宽
      minWidth:'200',//最小列宽
      canEdit:false,//是否显示可编辑单元格
      isClickEdit:false,//是否点击显示可编辑单元格
      editConfig:{},//编辑项配置
      filterFn:()=>{},//列过滤方法
      children:[],//多表头子列
      bind:{fixed:true,align:'right'} //第三方组件库属性
    },
]
tableConfig:{
    firstColumn:[ //第一列序号、复选框、单选框的配置,单独设置方便后续扩展
       {
        type:'index',
        label:'序号',
        width:'120',//列宽
        minWidth:'200',//最小列宽
        bind:{},//第三方组件属性
      }
    ],
    listTypeInfo:{ //可编辑表格的下拉框、复选框组、单选框组的选项配置
        hobbyList:[]
        cityList:[]
    },
}
